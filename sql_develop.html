SELECT
    a.enterprise_name,
    a.credit_code,
    b.industry_tag,
    c.change_item,
    c.change_date
FROM enterprise_base_info a
LEFT JOIN enterprise_tag_api b
    ON a.credit_code = b.credit_code
LEFT JOIN enterprise_change_history c
    ON a.credit_code = c.credit_code
WHERE a.enterprise_name LIKE '%亿云%'
ORDER BY c.change_date DESC
LIMIT 100;
